1. Install Apache server on Ubuntu
sudo apt install apache2

2. Install php runtime and php mysql connector
sudo apt install php libapache2-mod-php php-mysql

3. Install MySQL server
sudo apt install mysql-server 

4. Login to MySQL server
mysql -h <endpoint of the rds database> -u <username> -p <password>

5. Create a new database user for wordpress (change the password to something strong)
CREATE USER 'wp_user'@'%' IDENTIFIED BY 'Testpassword@123';

6. Create a database for wordpress
CREATE DATABASE wp;

7. Grant all privilges on the database 'wp' to the newly created user
GRANT ALL PRIVILEGES ON wp.* TO 'wp_user'@'%';

8. Apply privileges
FLUSH PRIVILEGES;

9. Download wordpress
apt install unzip
cd /tmp
wget https://wordpress.org/latest.zip

10. Unzip
unzip latest.zip

11. Move wordpress folder to apache document root
sudo mv wordpress/ /var/www/html

12. Command to restart/reload apache server
sudo systemctl restart apache2
OR
sudo systemctl reload apache2
